note
	description: "Storage backends for files."
	date: "$Date$"
	revision: "$Revision$"

class interface
	FILE_STORAGE_BACKEND

create 
	make_from_file,
	make_from_path,
	make_from_filename

feature -- Status setting

	open_read
			-- Open for reading.

	open_write
			-- Open for writing.

	close
			-- Close backend.
	
feature -- Status

	archive_finished: BOOLEAN
			-- Do the next two blocks contain only NUL bytes or file has not enough characters to read?

	block_ready: BOOLEAN
			-- Is there a block ready to read with last_block?

	readable: BOOLEAN
			-- Is Current open and readable?

	writable: BOOLEAN
			-- Is Current created and writable?

	closed: BOOLEAN
			-- Is Current closed?
	
feature -- Reading

	last_block: MANAGED_POINTER
			-- Return last block that was read.

	read_block
			-- Read next block.
	
feature -- Writing

	write_block (a_block: MANAGED_POINTER; a_pos: INTEGER_32)
			-- Write a_block, starting from a_pos.

	finalize
			-- Finalize archive (write two 0 blocks).
	
invariant
	buffer_size: block_buffer.count = {TAR_CONST}.tar_block_size
	buffer_entries_size: across
			buffer as ic
		all
			ic.item.count = {TAR_CONST}.tar_block_size
		end
